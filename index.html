Count the Heads

        var Count_Occ = (s) => 
        {
            let newString = "";
            let letters = "abcdefghijklmnopqrstuvwxyz";
            let count = 0;
            for (let letter of letters) {
                for (let character of s) {
                    if (letter === character) {
                        count++;
                    }
                }
                if (count > 1) {
                    newString += `${letter}${count}`;
                }
                count = 0;
            }
            return newString;
        };

Plaindrome Check

var Palin_Check = (str) =>
        {
            let rstring = "";
            let length=str.length;
            while (length > 0) {
                rstring += str[length-1];
                length--;
            }
            if (str === rstring) {
                return "True";
            }
            else{
                return "False";
            }
        }

Reverse the String

        var Reverse_String = (str) => 
        {
            let rstring = "";
            let length=str.length;
            while (length > 0) {
                rstring += str[length-1];
                length--;
            }
            return rstring;
        };

Count the Vowels and Consonants

        var Count_Vowels= (S) => 
        {
            S = S.toLowerCase();
            let vowels="aeiou" 
            let count=0;
            for (let letter of vowels) {
                for (let character of S) {
                    if (letter === character) {
                        count++;
                    }
                }
            }
            return count;    
        };
        var Count_Consonants= (S) => 
        {
            S = S.toLowerCase();
            let count=0;
            for (let letter of S) {
                if(letter !== "a" &&
                letter !== "e" &&
                letter !== "i" &&
                letter !== "o" &&
                letter !== "u") {
                    count++;
                }
            }
            return count;
        };
